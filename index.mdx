        ```typescript
        import HolyTime from 'holy-time'

        // Get the current time
        const now = HolyTime.now()

        // Chain operations fluently
        const future = HolyTime
          .in(4, 'days')
          .add(2, 'weeks')
          .subtract(4, 'minutes')

        // Check if it's the weekend
        console.log(future.isWeekend())
        ```
